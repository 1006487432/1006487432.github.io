## C.前缀平方和序列

求组合数。

找到x以内的平方数，按升序排列。

~~~ c++
#include<bits/stdc++.h>
using namespace std;
typedef long long ll;

const ll mod = 1e9 + 7;
ll fact[1000010];
ll n, x;
ll ans = 0;
ll ksm(ll a, ll b){
    ll res = 1;
    while(b){
        if(b & 1)    res = res * a % mod;
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}
ll inv(ll a){
    return ksm(a, mod - 2);
}
int main(){    
    fact[0] = 1;
    ll cnt = 0;
    cin >> n >> x;
    for(ll i = 1; i * i <= 1e6; i++){
        if(i * i <= x)    cnt++;
    }
    //cout << cnt << '\n';
    for(ll i = 1; i <= cnt; i++){
        fact[i] = fact[i - 1] * i % mod;
    }
    if(cnt < n){
        cout << 0;
        return 0;
    }
    ans = fact[cnt] * inv(fact[n]) % mod;
    ans = ans * inv(fact[cnt - n]) % mod;
    cout << ans;
    return 0;
}
~~~

## D.走一个大整数迷宫

~~~ C++
#include<bits/stdc++.h>

ll a[20][20];

int dx[] = {-1, 1, 0 ,0};

    ll x, y, val;
    node(ll x, ll y, ll val){
        this->x = x;
        this->y = y;
        this->val = val;
    }

    ll n, m, p;

        return 0;
    }
    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= m; j++)
            cin >> a[i][j], a[i][j] %= mod;
    for(int i = 1; i <= n; i++)
        for(int j = 1; j <= m; j++){
            ll temp;
            cin >> temp;    
        }
    queue<struct node> que;
    int cnt = 0;
    que.push(node(1, 1, a[1][1]));
    st[1][1][a[1][1]] = 1;
    while(que.size()){
        int len = que.size();
        while(len--){
            auto tmp = que.front();
            que.pop();
            int x = tmp.x, y = tmp.y, val = tmp.val;

                    cout << cnt + 1;
                    return 0;                    
                }
                que.push(node(xx,yy,val1));
            }    
        }
        cnt++;
    }
~~~

## E.前缀和前缀最大值

A类价值的最大、最小显然是数组降序、升序排列时的价值。不难发现A类价值是连续的，因此问题转移到如何快速求B类价值，上界显然就是正数的个数+1，下界则是负数全放前面，正数从小到大排列。由于B的值域很小，可以前缀和求出每个子数组中每个正数的个数。求下界时，只需要知道花费多少个正数将负数全部抵消即可。

~~~ c++
#include <bits/stdc++.h>
using namespace std;
void solve()
{
	int n;
	cin >> n;
	vector<int> a(n + 1), pre_neg(n + 1);
	vector<vector<int>> cnt_pos(n + 1, vector<int>(110));
	for (int i = 1; i <= n; i ++ )
	{
		cin >> a[i];
		if(a[i] < 0)    pre_neg[i] = pre_neg[i - 1] + a[i];
        else pre_neg[i] = pre_neg[i - 1];
		for (int j = 1; j <= 100; j ++ )
		{
			cnt_pos[i][j] = cnt_pos[i - 1][j] + (a[i] == j);
		}
	} 
	int q;
	cin >> q;
	while (q -- )
	{
		int l, r;
		cin >> l >> r;
        int mx = 0;
		for(int i = 1; i <= 100; i++) mx += cnt_pos[r][i] - cnt_pos[l - 1][i];
        int all = pre_neg[r] - pre_neg[l - 1];
        int need = 0;
        for(int i = 1; i <= 100; i++){
            int cnt = cnt_pos[r][i] - cnt_pos[l - 1][i];
            if(all + cnt * i >= 0){
                need += (abs(all)) / i;
                break;
            }else{
                all += cnt * i;
                need += cnt;
            }
        }
        cout << need + 1 << '\n';
	}
}

int main()
{
	ios::sync_with_stdio(false);
	cin.tie(0), cout.tie(0);
	int t = 1;
	// cin >> t;
	while (t--)
	{
		solve();
	}
}


~~~

## F.parent 树上启发式合并

保证所有询问的不同的字符串 T 的长度和不超过 10^4。说明不同字符串的长度是根号级别的，$\sqrt{(10^4)}$。将询问离线后，按长度枚举字串，使用字符串哈希进行匹配并记录匹配次数，若匹配且次数刚好是询问的次数，则记录下这个答案。

~~~ c++
#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long ull;
const int N = 100010;
ull h[N], p[N];
ull base = 131;
map<pair<ull, int>, int> mp;//存储答案
vector<pair<ull, int> > ask;//离线询问
map<ull, set<int> > has;//询问串的询问次数
map<int, set<ull>> num;//按长度存询问串的哈希
int main(){
    int n, q;
    cin >> n >> q;
    string s;
    cin >> s;
    p[0] = 1;
    for(int i = 1; i <= n; i++){
        h[i] = h[i - 1] * base + s[i - 1];
        p[i] = p[i - 1] * base;
    }
    auto get = [&](int l, int r){
        return h[r] - h[l - 1] * p[r - l + 1];
    };
    auto get_hash = [&](string tmp){
        ull hs = 0;
        for(auto i : tmp)    hs = hs * base + i;
        return hs;
    };
    for(int i = 1; i <= q; i++){
        string tmp;
        int k;
        cin >> tmp >> k;
        auto hs = get_hash(tmp);
        ask.push_back({hs, k});
        has[hs].insert(k);
        num[tmp.size()].insert(hs);
    }
    for(auto [len, st] : num){
        map<ull,int> ump;
        for(int i = len; i <= n; i++){
            ull myhash = get(i - len + 1, i);
            
            if(num[len].find(myhash) != num[len].end()){
                ump[myhash]++;
                if(has[myhash].find(ump[myhash]) != has[myhash].end()){
                    mp[{myhash, ump[myhash]}] = i;
                }
            }
        }
    }
    for(auto [hs, k] : ask){
        if(!mp[{hs,k}]){
            cout << -1 << '\n';
        }else{
            cout << mp[{hs,k}] << '\n';
        }
    }
    return 0;
}	
~~~
