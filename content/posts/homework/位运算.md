# 位运算

## 模块编写
### 头文件声明

~~~ c++
#include<stdio.h>
#include<time.h>
~~~

### 函数、变量声明

~~~ c++
unsigned short TIME;
void setHour(int h);
void setMinute(int m);
void setSecond(int s);
void getTime();
void printTime();
~~~

### 函数编写

~~~ c++
void getTime(){
    time_t currentTime;
    struct tm *localTime;
    time(&currentTime);
    localTime = localtime(&currentTime);
    //6 6 5
    //h m s
    setHour(localTime->tm_hour);
    setMinute(localTime->tm_min);
    setSecond(localTime->tm_sec);
}
void printTime(){
    printf("Now is: %d:%d:%d\n", TIME >> 11, (TIME & (0x3f << 5)) >> 5, TIME & 0x1f);
}
void setHour(int h){
    unsigned int th = h & 0x3f;
    th <<= 11;
    TIME &= ~(0x3f << 11);
    TIME |= th;
}
void setMinute(int m){
    unsigned int tm = m & 0x3f;
    tm <<= 5;
    TIME &= ~(0x3f << 5);
    TIME |= tm;
}
void setSecond(int s){
    unsigned int ts = s & 0x3f;
    TIME &= ~(0x1f);
    TIME |= ts;
}
~~~

## 模块测试

### 测试代码

~~~ c++
int main(){
    getTime();
    printTime();
    printf("update:");
    setHour(20);
    setMinute(20);
    setSecond(20);
    printTime();
    printf("update Minute:");
    setMinute(15);
    printTime();
    printf("Minute is:%d\n",  (TIME & (0x3f << 5)) >> 5);
}
~~~
