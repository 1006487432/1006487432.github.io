typedef long long ll;
ll fpow(ll a, ll b, ll p)
{
	ll res = 1;
	while (b)
	{
		if (b & 1)	res = res * a % p;
		a = a * a % p;
		b >>= 1;
	}
	return res;
}
ll bsgs(ll a, ll b, ll p)
{
	a %= p, b %= p;
	if(a == 0) return b == 0 ? 1 : -1; 
	if(b == 1) return 0;
	ll m = ceil(sqrt(p)), inv = fpow(a, m, p);
	unordered_map<ll, ll> mp;
	mp[b] = 0;
	ll tmp = inv;
	for (int i = 1, e = b; i < m; i++)
	{
		e = e * a % p;
		mp[e] = i;
	}
	for (int i = 1; i <= m; i++)
	{
		if (mp.find(inv) != mp.end())
		{
			return i * m - mp[inv];
		}
		inv = inv * tmp % p;
	}
	return -1;
}

{
